" Set the font
"set guifont=JetBrainsMono\ Nerd\ Font\ Regular:h15

" Disable compatibility with vi which can cause unexpected issues.
set nocompatible

" Enable type file detection. Vim will be able to try to detect the type of file in use.
filetype on

" Enable plugins and load plugin for the detected file type.  filetype plugin on

" Load an indent file for the detected file type.
filetype indent on


" Turn syntax highlighting on.
syntax on

" Turn line numbers.
set number

" Turn on relative line numbers.
set relativenumber

" Treat all numerals as decimal
" set nrformats

" Compile and run C file with F9
autocmd FileType c nnoremap <F9> :w<CR>:!gcc -Wall -Wextra -Werror -Wshadow -std=c11 -g % -o %< && ./%<<CR>

" Compile and run C++ file with F9
autocmd FileType cpp nnoremap <F9> :w<CR>:!g++ -std=c++20 -Wall -Wextra -pedantic -O2 -g % -o %< && ./%<<CR>

" Run Python files with F9
autocmd FileType python nnoremap <F9> :w<CR>:!python3 %<CR>

" Run Cargo for rust with F9
autocmd FileType rust nnoremap <F9> :w<CR>:!cargo run<CR>
" Run RustC for rust with F10
autocmd FIleType rust nnoremap <F10> :w<CR>:!rustc % && ./%<<CR>

" Run Zig Build for Zig files with F9
autocmd FileType zig nnoremap <F9> :w<CR>:!zig run %<CR>

" Run Zig Test for Zig files with F10
autocmd FileType zig nnoremap <F10> :w<CR>:!zig test %<CR>
